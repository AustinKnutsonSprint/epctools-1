76  EIpAddress &operator=(cpStr addr)
77  {
78  auto parts = EUtility::split(addr, "/");
79  memset(&address_, 0, sizeof(address_));
80  if (inet_pton(AF_INET, parts[0], &address_.ipv4) == 1)
81  {
82  if (family_ != AF_INET)
83  family_ = AF_INET;
84  maskWidth_ = (parts.size() > 1) ? static_cast<UChar>(std::stoi(parts[1])) : 32;
85  }
86  else if (inet_pton(AF_INET6, parts[0], &address_.ipv6) == 1)
87  {
88  if (family_ != AF_INET6)
89  family_ = AF_INET6;
90  maskWidth_ = (parts.size() > 1) ? static_cast<UChar>(std::stoi(parts[1])) : 128;
91  }
92  else
93  {
94  family_ = AF_UNSPEC;
95  maskWidth_ = 0;
96  }
97 
98  return *this;
99  }

103  EIpAddress &operator=(const EIpAddress &ipaddr)
104  {
105  family_ = ipaddr.family_;
106  maskWidth_ = ipaddr.maskWidth_;
107  memset(&address_, 0, sizeof(address_));
108  switch (family_)
109  {
110  case AF_INET: { address_.ipv4.s_addr = ipaddr.address_.ipv4.s_addr; break; }
111  case AF_INET6: { memcpy(address_.ipv6.s6_addr, ipaddr.address_.ipv6.s6_addr, sizeof(address_.ipv6.s6_addr)); break; }
112  default: { break; }
113  }
114  return *this;
115  }

119  EIpAddress &operator=(const struct sockaddr_storage &saddr)
120  {
121  switch (saddr.ss_family)
122  {
123  case AF_INET: { return EIpAddress::operator=(((const struct sockaddr_in&)saddr).sin_addr); }
124  case AF_INET6: { return EIpAddress::operator=(((const struct sockaddr_in6&)saddr).sin6_addr); }
125  default:

194  std::string address(Bool includeMask = True) const
195  {
196  if (isAny())
197  {
198  std::string str = "any";
199  return str;
200  }
201 
202  Char buf[INET6_ADDRSTRLEN] = {};
203 
204  switch (family_)
205  {
206  case AF_INET: { inet_ntop(AF_INET,&address_.ipv4,buf,sizeof(buf)); break; }
207  case AF_INET6: { inet_ntop(AF_INET6,&address_.ipv6,buf,sizeof(buf)); break; }
208  default: { break; }
209  }
210 
211  std::stringstream ss;
212  ss << buf;
213  if (includeMask)
214  ss << "/" << static_cast<Int>(maskWidth());
215 
216  return ss.str();
217  }

267  Bool isAny() const
268  {
269  if (maskWidth_ != 0)
270  return False;
271  switch (family_)
272  {
273  case AF_INET: { if (address_.ipv4.s_addr == 0) return True; break; }
274  case AF_INET6: { for (auto b : address_.ipv6.s6_addr) { if (b != 0) return False; } return True; }
275  default: { break; }
276  }
277  return False;
278  }

295  std::size_t operator()(const EIpAddress &addr) const noexcept
296  {
297  size_t iphash = 0xc70f6907UL;
298  switch (addr.family())
299  {
300  case AF_INET: { iphash = EMurmurHash64::getHash(reinterpret_cast<cpUChar>(&addr.ipv4Address()), sizeof(in_addr)); break; }
301  case AF_INET6: { iphash = EMurmurHash64::getHash(reinterpret_cast<cpUChar>(&addr.ipv6Address()), sizeof(in6_addr)); break; }
302  default: { break; }
303  }
304  size_t mwhash = std::hash<UChar>{}(addr.maskWidth());
305  return EMurmurHash64::combine(iphash, mwhash);
306  }
